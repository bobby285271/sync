void dfs(ll curn, ll index)
{
    if (index == m - 1)
    {
        if (index % 2 == 0)
        {
            ans += n - curn;
            ans %= 10000;

        else
        {
            ans += curn;

        }

        {

                dfs(i, index + 1);
            }
        }
    }
